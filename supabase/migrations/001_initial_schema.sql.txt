CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

CREATE TABLE public.profiles (
  id UUID REFERENCES auth.users ON DELETE CASCADE PRIMARY KEY,
  full_name TEXT,
  avatar_url TEXT,
  conditions JSONB DEFAULT '[]',
  allergies TEXT[] DEFAULT '{}',
  dietary_restrictions TEXT[] DEFAULT '{}',
  age INTEGER,
  weight INTEGER,
  height INTEGER,
  gender TEXT,
  activity_level TEXT,
  daily_budgets JSONB DEFAULT '{"sodium_mg": 2300, "sugar_g": 50, "saturated_fat_g": 20, "caffeine_mg": 400, "alcohol_g": 0, "custom_limits": {}}',
  family_members JSONB DEFAULT '[]',
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

ALTER TABLE public.profiles ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Users can view own profile" ON public.profiles FOR SELECT USING (auth.uid() = id);
CREATE POLICY "Users can update own profile" ON public.profiles FOR UPDATE USING (auth.uid() = id);
CREATE POLICY "Users can insert own profile" ON public.profiles FOR INSERT WITH CHECK (auth.uid() = id);

CREATE TABLE public.scans (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  user_id UUID REFERENCES auth.users ON DELETE CASCADE NOT NULL,
  barcode TEXT NOT NULL,
  product_name TEXT,
  product_image TEXT,
  brands TEXT,
  health_score INTEGER,
  risk_level TEXT,
  additives JSONB DEFAULT '[]',
  nutritional_warnings JSONB DEFAULT '[]',
  personalized_risks JSONB DEFAULT '[]',
  daily_budget_impact JSONB DEFAULT '{}',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX idx_scans_user_id ON public.scans(user_id);
CREATE INDEX idx_scans_created_at ON public.scans(created_at DESC);
ALTER TABLE public.scans ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own scans" ON public.scans FOR ALL USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id);

CREATE TABLE public.daily_logs (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  user_id UUID REFERENCES auth.users ON DELETE CASCADE NOT NULL,
  log_date DATE NOT NULL,
  sodium_mg INTEGER DEFAULT 0,
  sugar_g INTEGER DEFAULT 0,
  saturated_fat_g INTEGER DEFAULT 0,
  protein_g INTEGER DEFAULT 0,
  fiber_g INTEGER DEFAULT 0,
  calories INTEGER DEFAULT 0,
  additives_count INTEGER DEFAULT 0,
  ultra_processed_score INTEGER DEFAULT 0,
  scan_count INTEGER DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, log_date)
);

CREATE INDEX idx_daily_logs_user_date ON public.daily_logs(user_id, log_date DESC);
ALTER TABLE public.daily_logs ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own daily logs" ON public.daily_logs FOR ALL USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id);

CREATE TABLE public.chemical_exposure (
  id UUID DEFAULT uuid_generate_v4() PRIMARY KEY,
  user_id UUID REFERENCES auth.users ON DELETE CASCADE NOT NULL,
  chemical_name TEXT NOT NULL,
  chemical_code TEXT NOT NULL,
  amount_mg NUMERIC DEFAULT 0,
  week_start DATE NOT NULL,
  source_products JSONB DEFAULT '[]',
  warning_triggered BOOLEAN DEFAULT FALSE,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(user_id, chemical_code, week_start)
);

CREATE INDEX idx_chemical_exposure_user_week ON public.chemical_exposure(user_id, week_start DESC);
ALTER TABLE public.chemical_exposure ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Users can manage own chemical exposure" ON public.chemical_exposure FOR ALL USING (auth.uid() = user_id) WITH CHECK (auth.uid() = user_id);

CREATE OR REPLACE FUNCTION update_updated_at_column() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = NOW(); RETURN NEW; END; $$ language 'plpgsql';
CREATE TRIGGER update_profiles_updated_at BEFORE UPDATE ON public.profiles FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE OR REPLACE FUNCTION public.handle_new_user() RETURNS TRIGGER AS $$ BEGIN INSERT INTO public.profiles (id, full_name, avatar_url) VALUES (NEW.id, NEW.raw_user_meta_data->>'full_name', NEW.raw_user_meta_data->>'avatar_url'); RETURN NEW; END; $$ LANGUAGE plpgsql SECURITY DEFINER;
CREATE TRIGGER on_auth_user_created AFTER INSERT ON auth.users FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
